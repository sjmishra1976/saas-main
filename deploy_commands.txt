# 0) Vars
export PROJECT_ID="saas-main-vishusystems"
export REGION="us-west1"
export CLUSTER="vishu-aiaas"
export NAMESPACE="saas"
export DOMAIN="vishuaiaas.com"
export SA_NAME="saas-gke-deployer"
export SA_EMAIL="${SA_NAME}@${PROJECT_ID}.iam.gserviceaccount.com"
export KEY_FILE="$HOME/.gcp-keys/gcp-sa.json"

# 1) Auth + project
gcloud auth login
gcloud config set project "$PROJECT_ID"

# 2) Ensure cluster exists (skip if already created)
gcloud container clusters create-auto "$CLUSTER" \
  --project="$PROJECT_ID" \
  --region="$REGION" \
  --release-channel=regular

# 2.5) Install plugin using command: gcloud components install gke-gcloud-auth-plugin

# 3) Configure kubectl
gcloud container clusters get-credentials "$CLUSTER" \
  --region "$REGION" \
  --project "$PROJECT_ID"

# 4) Create service account for API-driven GKE deploys
gcloud iam service-accounts create "$SA_NAME" \
  --display-name="SaaS GKE Deployer" || true

gcloud projects add-iam-policy-binding "$PROJECT_ID" \
  --member="serviceAccount:${SA_EMAIL}" \
  --role="roles/container.admin"

gcloud projects add-iam-policy-binding "$PROJECT_ID" \
  --member="serviceAccount:${SA_EMAIL}" \
  --role="roles/compute.loadBalancerAdmin"

# Optional if you want to manage Cloud DNS via gcloud
gcloud projects add-iam-policy-binding "$PROJECT_ID" \
  --member="serviceAccount:${SA_EMAIL}" \
  --role="roles/dns.admin"

# 5) Generate SA key file
mkdir -p "$(dirname "$KEY_FILE")"
gcloud iam service-accounts keys create "$KEY_FILE" \
  --iam-account="$SA_EMAIL"
chmod 600 "$KEY_FILE"

# 6) Reserve global static IP for ingress
gcloud compute addresses create saas-ingress-ip \
  --global \
  --project "$PROJECT_ID" || true

export INGRESS_IP=$(gcloud compute addresses describe saas-ingress-ip \
  --global \
  --project "$PROJECT_ID" \
  --format='value(address)')
echo "INGRESS_IP=$INGRESS_IP"

# 7) Register domain in Cloud Domains (if not already registered)
# (UI step typically; skip if already owned)
# https://console.cloud.google.com/net-services/domains/registrations

# 8) Cloud DNS public zone (if not already created)
gcloud dns managed-zones create vishuaiaas-zone \
  --dns-name="${DOMAIN}." \
  --description="vishuaiaas public zone" \
  --project "$PROJECT_ID" || true

# 9) Show nameservers and set these at domain registrar
gcloud dns managed-zones describe vishuaiaas-zone \
  --project "$PROJECT_ID" \
  --format="value(nameServers)"

# 10) Create DNS A records to ingress IP
gcloud dns record-sets create "app.${DOMAIN}." \
  --zone="vishuaiaas-zone" \
  --type="A" \
  --ttl="300" \
  --rrdatas="$INGRESS_IP" \
  --project "$PROJECT_ID" || true

gcloud dns record-sets create "api.${DOMAIN}." \
  --zone="vishuaiaas-zone" \
  --type="A" \
  --ttl="300" \
  --rrdatas="$INGRESS_IP" \
  --project "$PROJECT_ID" || true

# 11) Build/push images (Docker Hub example)
docker login
docker build -t <dockerhub-username>/saas-api:latest -f apps/api/Dockerfile.prod .
docker build -t <dockerhub-username>/saas-web:latest -f apps/web/Dockerfile .
docker push <dockerhub-username>/saas-api:latest
docker push <dockerhub-username>/saas-web:latest

# 12) Patch k8s image refs to your Docker Hub repo
# edit files:
# k8s/api.yaml -> image: <dockerhub-username>/saas-api:latest
# k8s/web.yaml -> image: <dockerhub-username>/saas-web:latest

# 13) Create namespace first
kubectl apply -f k8s/namespace.yaml

# 14) Create docker pull secret (only if Docker Hub images are private)
kubectl -n "$NAMESPACE" create secret docker-registry dockerhub-creds \
  --docker-server=https://index.docker.io/v1/ \
  --docker-username=<dockerhub-username> \
  --docker-password=<dockerhub-access-token> \
  --docker-email=<your-email> || true


# Note: you can use kubectl apply -f using secret.yaml except sa.json to avoid copying json to secret.yaml, use below command after other secrets created
kubectl -n saas create secret generic saas-secrets \
  --from-file=GCP_SERVICE_ACCOUNT_KEY_JSON=$HOME/.gcp-keys/gcp-sa.json \
  --dry-run=client -o yaml | kubectl apply -f -

If you wish directly doing for other key can use below coammnd also

# 15) Create app secret from real values + SA key
kubectl -n "$NAMESPACE" create secret generic saas-secrets \
  --from-literal=AUTH0_DOMAIN="your-tenant.us.auth0.com" \
  --from-literal=AUTH0_CLIENT_ID="your-client-id" \
  --from-literal=AUTH0_CLIENT_SECRET="your-client-secret" \
  --from-literal=AUTH0_SECRET="your-long-random-secret" \
  --from-literal=MONGODB_URI="your-mongodb-uri" \
  --from-literal=N8N_DB_PASSWORD="your-n8n-db-password" \
  --from-literal=N8N_ENCRYPTION_KEY="your-n8n-encryption-key" \
  --from-literal=SERVICE_ACTIVITY_INGEST_KEY="your-activity-ingest-key" \
  --from-file=GCP_SERVICE_ACCOUNT_KEY_JSON="$KEY_FILE" \
  --dry-run=client -o yaml | kubectl apply -f -

# 16) Update configmap values (domains/auth audience/base URLs), then apply
kubectl apply -f k8s/configmap.yaml

# 17) Apply workloads + cert + ingress
kubectl apply -f k8s/mongo.yaml
kubectl apply -f k8s/postgres.yaml
kubectl apply -f k8s/api.yaml
kubectl apply -f k8s/web.yaml
kubectl apply -f k8s/managed-cert.yaml
kubectl apply -f k8s/ingress.yaml

# 18) Verify
kubectl -n "$NAMESPACE" get pods
kubectl -n "$NAMESPACE" get svc
kubectl -n "$NAMESPACE" get ingress
kubectl -n "$NAMESPACE" get managedcertificate
kubectl -n "$NAMESPACE" describe managedcertificate saas-cert
